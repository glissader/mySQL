# В базе данных Redis подберите коллекцию для подсчета посещений с определенных IP-адресов.
set "127.0.0.1" 1 # первый визит
incr "127.0.0.1" # последующие визиты
incr "127.0.0.1"
get "127.0.0.1" # количество посещений

# При помощи базы данных Redis решите задачу поиска имени пользователя по электронному адресу
# e-mail адреса уникальны
set "e1@mail" "name"
set "e2@mail" "name"
set "e3@mail" "name"
get "e1@mail"

# и наоборот, поиск электронного адреса пользователя по его имени.
# имена могут повторяться
sadd "name" "e1@mail"
sadd "name" "e2@mail"
sadd "name" "e3@mail"
smembers "name"
